import Calendar from 'buck3000-react-infinite-calendar';
import 'buck3000-react-infinite-calendar/dist/styles.css';

function MyCalendarComponent() {
  const [selected, setSelected] = useState(new Date());
  
  return (
    <Calendar
      width={400}
      height={500}
      selected={selected}
      onSelect={setSelected}
      minDate={new Date(2020, 0, 1)}
      maxDate={new Date(2030, 11, 31)}
    />
  );
}

Rails Integration Instructions

1. Installation

# Add to your package.json
npm install buck3000-react-infinite-calendar

# Or with yarn
yarn add buck3000-react-infinite-calendar

2. Import in your Rails app

// app/javascript/components/calendar.jsx
import React, { useState } from 'react';
import Calendar from 'buck3000-react-infinite-calendar';
import 'buck3000-react-infinite-calendar/dist/buck3000-react-infinite-calendar-fork.css';

export default function MyCalendar(props) {
  const [selected, setSelected] = useState(props.initialDate || new Date());
  
  const handleSelect = (date) => {
    setSelected(date);
    // Send to Rails backend if needed
    if (props.onChange) {
      props.onChange(date);
    }
  };
  
  return (
    <Calendar
      width={Math.min(window.innerWidth - 40, 400)}
      height={500}
      selected={selected}
      onSelect={handleSelect}
      minDate={props.minDate ? new Date(props.minDate) : new Date(2020, 0, 1)}
      maxDate={props.maxDate ? new Date(props.maxDate) : new Date(2030, 11, 31)}
      locale={props.locale}
      theme={props.theme}
    />
  );
}

3. Use in Rails views

# app/views/events/_calendar.html.erb
<%= react_component "MyCalendar", {
  initialDate: @event&.date&.iso8601,
  minDate: Date.current.iso8601,
  maxDate: 1.year.from_now.iso8601,
  locale: {
    weekdays: %w[Sun Mon Tue Wed Thu Fri Sat],
    months: Date::MONTHNAMES.compact
  },
  onChange: "handleDateChange"
} %>

4. Handle data in Rails

# app/controllers/events_controller.rb
def update_date
  @event = Event.find(params[:id])
  @event.update(date: params[:selected_date])
  render json: { status: 'success' }
end
